import { Request, Response } from 'express';
import { InvoiceModel } from '../models/invoice.model';
import { PaymentService } from '../services/paymentService';
import { ApiError } from '../utils/ApiError';
import { handleError } from '../utils/errorHandler';
import { EmailQueueService } from '../services/emailQueueService';

export class InvoiceController {
  private emailQueueService: EmailQueueService;
  private paymentService: PaymentService;

  constructor() {
    this.emailQueueService = new EmailQueueService();
    this.paymentService = new PaymentService();
  }

  createInvoice = async (req: Request, res: Response) => {
    try {
      const invoice = await InvoiceModel.create(req.body);
      
      // Queue invoice email if customer email exists
      if (invoice.customer?.email) {
        await this.emailQueueService.addToQueue({
          type: 'invoice',
          invoiceId: invoice.id,
          recipientEmail: invoice.customer.email
        });
      }

      res.status(201).json({ success: true, data: invoice });
    } catch (error) {
      handleError(error, res, 'Error creating invoice');
    }
  }

  recordPayment = async (req: Request, res: Response) => {
    try {
      const invoice = await this.paymentService.recordPayment(req.params.id, req.body);
      
      // Queue payment confirmation email
      await this.emailQueueService.addToQueue({
        type: 'confirmation',
        invoiceId: invoice.id
      });

      res.json({ success: true, data: invoice });
    } catch (error) {
      handleError(error, res, 'Error recording payment');
    }
  }

  getAllInvoices = async (req: Request, res: Response) => {
    try {
      const invoices = await InvoiceModel.find();
      res.json({ success: true, data: invoices });
    } catch (error) {
      handleError(error, res, 'Error fetching invoices');
    }
  }

  getInvoiceById = async (req: Request, res: Response) => {
    try {
      const invoice = await InvoiceModel.findOne({ id: req.params.id });
      if (!invoice) throw new ApiError(404, 'Invoice not found');
      res.json({ success: true, data: invoice });
    } catch (error) {
      handleError(error, res, 'Error fetching invoice');
    }
  }

  updateInvoice = async (req: Request, res: Response) => {
    try {
      const invoice = await InvoiceModel.findOneAndUpdate(
        { id: req.params.id },
        req.body,
        { new: true }
      );
      if (!invoice) throw new ApiError(404, 'Invoice not found');
      res.json({ success: true, data: invoice });
    } catch (error) {
      handleError(error, res, 'Error updating invoice');
    }
  }

  deleteInvoice = async (req: Request, res: Response) => {
    try {
      const invoice = await InvoiceModel.findOneAndDelete({ id: req.params.id });
      if (!invoice) throw new ApiError(404, 'Invoice not found');
      res.json({ success: true, message: 'Invoice deleted successfully' });
    } catch (error) {
      handleError(error, res, 'Error deleting invoice');
    }
  }

  getPaymentStatus = async (req: Request, res: Response) => {
    try {
      const invoice = await InvoiceModel.findOne({ id: req.params.id });
      if (!invoice) throw new ApiError(404, 'Invoice not found');
      if (!invoice.total) throw new ApiError(400, 'Invoice total is not set');
      
      const payments = await this.paymentService.getPaymentsByInvoiceId(req.params.id);
      const paidAmount = payments.reduce((sum, payment) => sum + payment.amount, 0);
      
      res.json({
        success: true,
        data: {
          status: invoice.status,
          paidAmount: paidAmount,
          remainingAmount: invoice.total - paidAmount,
          dueDate: invoice.dueDate
        }
      });
    } catch (error) {
      handleError(error, res, 'Error fetching payment status');
    }
  }
}
